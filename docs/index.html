import fitz  # PyMuPDF
import os

def pdf_to_images(pdf_path, output_dir="images", dpi=300):
    """
    Converts PDF to images using PyMuPDF (Fitz).
    Yields image paths.
    """
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)

    doc = fitz.open(pdf_path)
    image_paths = []

    for page_num in range(len(doc)):
        page = doc.load_page(page_num)
        
        # Matrix for scaling (DPI / 72 is the scale factor since default is 72)
        zoom = dpi / 72 
        mat = fitz.Matrix(zoom, zoom)
        
        pix = page.get_pixmap(matrix=mat)
        img_path = os.path.join(output_dir, f"page_{page_num + 1}.png")
        pix.save(img_path)
        image_paths.append(img_path)

    doc.close()
    return image_paths

    else:
        print(f"File {pdf_file} not found.")

def run_ocr_json(image_path):
    """
    Implementation of Recommendation 1: Spatial JSON Payload.
    Preserves text and bounding box coordinates for the LLM.
    """
    img = load_image(image_path)
    result = ocr.ocr(img, cls=True)

    ocr_payload = []

    if result and result[0]:
        for line in result[0]:
            bbox = line[0]              # [[x1,y1],[x2,y2],[x3,y3],[x4,y4]]
            text = line[1][0]
            conf = float(line[1][1])

            # Simplify bbox to [x_min, y_min, x_max, y_max]
            xs = [p[0] for p in bbox]
            ys = [p[1] for p in bbox]
            
            ocr_payload.append({
                "text": text,
                "bbox": [min(xs), min(ys), max(xs), max(ys)],
                "confidence": round(conf, 4)
            })

    return ocr_payload, img

def visualize_ocr(img, blocks):
    """Shows the bounding boxes plotted over the original image."""

    for b in blocks:
        x1, y1 = b["x_min"], b["y_min"]

            plt.Rectangle(
                (x1, y1),
                x2 - x1,
                y2 - y1,
                fill=False,

        )
    
    plt.axis("off")
    plt.title("OCR Bounding Boxes (Markdown Mode)")
    plt.show()

def group_into_lines(ocr_data, row_threshold=15):
    """
    Groups OCR blocks into virtual 'lines' based on their Y-coordinate.
    Implementation of Recommendation 2: Markdown/Layout Reconstruction.
    """
    ocr_data = sorted(ocr_data, key=lambda x: x["y_min"])
    
    lines = []
    for item in ocr_data:
        placed = False
        item_y = item["y_min"]
        
        for line in lines:
            line_y = line[0]["y_min"]
            if abs(item_y - line_y) < row_threshold:
                line.append(item)
                placed = True
                break
        
        if not placed:
            lines.append([item])
    
    reconstructed_text = []
    for line in lines:
        row_sorted = sorted(line, key=lambda x: x["x_min"])
        row_content = "   |   ".join(c["text"] for c in row_sorted)
        reconstructed_text.append(f"| {row_content} |")

    return "\n".join(reconstructed_text)

def run_ocr_markdown(image_path):
    img = load_image(image_path)
    result = ocr.ocr(img, cls=True)

    ocr_blocks = []
    if result and result[0]:
        for line in result[0]:
            bbox = line[0]
            text = line[1][0]
            
            xs = [p[0] for p in bbox]
            ys = [p[1] for p in bbox]
            
            ocr_blocks.append({
                "text": text,
                "x_min": min(xs),
                "y_min": min(ys),
                "x_max": max(xs),
                "y_max": max(ys)
            })

    markdown_layout = group_into_lines(ocr_blocks)
    return markdown_layout, ocr_blocks, img

from paddleocr import PPStructure, PaddleOCR
import os
import cv2
import numpy as np
import matplotlib.pyplot as plt
from PIL import Image
import json

# Initialize BOTH engines for a Hybrid Approach
# 1. Standard OCR (Best for finding every word, including the name)
ocr_standard = PaddleOCR(use_angle_cls=True, lang='en', show_log=False)
# 2. Structural Engine (Best for tables and headers)
layout_engine = PPStructure(table=True, ocr=True, layout=True, show_log=False)

def load_image(path):
    img = cv2.imread(path)
    return cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

def detect_checkboxes_opencv(img_gray, all_text_bboxes):
    """
    Fixed detector: Uses a combined mask from BOTH engines.
    """
    mask = np.zeros_like(img_gray)
    for bbox in all_text_bboxes:
        x1, y1, x2, y2 = [int(v) for v in bbox]
        cv2.rectangle(mask, (x1-5, y1-5), (x2+5, y2+5), 255, -1)

    _, binary = cv2.threshold(img_gray, 200, 255, cv2.THRESH_BINARY_INV)
    binary[mask == 255] = 0 # Wipe out all detected text before finding checkboxes

    contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    
    checkboxes = []
    for cnt in contours:
        x, y, w, h = cv2.boundingRect(cnt)
        aspect_ratio = float(w)/h
        if 16 < w < 40 and 16 < h < 40 and 0.8 < aspect_ratio < 1.2:
            roi = binary[y:y+h, x:x+w]
            density = cv2.countNonZero(roi) / (w * h)
            checkboxes.append({
                "type": "checkbox",
                "bbox": [float(x), float(y), float(x+w), float(y+h)],
                "is_checked": bool(density > 0.25)
            })
    return checkboxes

def run_production_ocr(image_path):
    img = load_image(image_path)
    img_gray = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    
    # STEP 1: Run Standard OCR to ensure we get the Name (and all other text)
    standard_res = ocr_standard.ocr(img)
    
    # STEP 2: Run Layout Engine for Tables
    structure_res = layout_engine(img)
    
    final_blocks = []
    all_text_bboxes = []

    # Process Standard OCR results first (High reliability for text)
    if standard_res and standard_res[0]:
        for line in standard_res[0]:
            bbox_raw = line[0] # [[x1,y1],[x2,y2],[x3,y3],[x4,y4]]
            text = line[1][0]
            conf = float(line[1][1])
            xs, ys = [p[0] for p in bbox_raw], [p[1] for p in bbox_raw]
            bbox = [min(xs), min(ys), max(xs), max(ys)]
            
            final_blocks.append({
                "type": "text",
                "text": text,
                "confidence": conf,
                "bbox": bbox
            })
            all_text_bboxes.append(bbox)

    # Process Layout Engine for Tables only (Structure)
    for block in structure_res:
        if block["type"] == "table":
            bbox = [float(c) for c in block["bbox"]]
            final_blocks.append({
                "type": "table",
                "bbox": bbox,
                "html": block["res"].get("html", "")
            })
            all_text_bboxes.append(bbox)
                
    # STEP 3: Detect Checkboxes (using the mask from all detected text)
    checkboxes = detect_checkboxes_opencv(img_gray, all_text_bboxes)
    final_blocks.extend(checkboxes)
    
    return final_blocks, img

def visualize_production(img, blocks):
    plt.figure(figsize=(12, 16))
    plt.imshow(img)
    
    for b in blocks:
        if b["type"] == "checkbox":
            color = "blue" if b["is_checked"] else "gray"
        elif b["type"] == "table":
            color = "red"
        else:
            color = "orange" if b.get("manual_review") else "green"
            
        bbox = b["bbox"]
        if len(bbox) == 4:
            x1, y1, x2, y2 = bbox
        else: # flattened polygon
            x1, y1, x2, y2 = bbox[0], bbox[1], bbox[4], bbox[5]

        plt.gca().add_patch(plt.Rectangle((x1, y1), x2-x1, y2-y1, fill=False, edgecolor=color, linewidth=1))

    plt.axis("off")
    plt.title("Production Pipeline (Red=Table, Green=Text, Blue=Checked, Gray=Empty, Orange=Review)")
    plt.show()

if __name__ == "__main__":
    photos_dir = "Photos"
    final_output = {}

    if not os.path.exists(photos_dir):
        print(f"Error: {photos_dir} directory not found.")
    else:
        # Sort and process page-by-page
        image_files = sorted([f for f in os.listdir(photos_dir) if f.endswith(".png")])
        
        for img_name in image_files:
            print(f"[INFO] Running Structural Analysis on {img_name}...")
            image_path = os.path.join(photos_dir, img_name)
            
            blocks, img_np = run_production_ocr(image_path)
            final_output[img_name] = blocks
            
            # Interactive visualization
            visualize_production(img_np, blocks)

        # High-Fidelity JSON Output
        output_file = "ocr_production_data.json"
        with open(output_file, "w", encoding="utf-8") as f:
            json.dump(final_output, f, indent=2)
        
        print(f"\n✅ Production Pipeline Complete: Saved complex structural data to {output_file}")
cleaner.py 
import json
from collections import defaultdict

# ----------------------------
# CONFIG
# ----------------------------
CONF_THRESHOLD = 0.7
Y_THRESHOLD = 20   # line merge threshold

SECTION_KEYS = [
    "section", "borrower", "employment", "address",
    "income", "contact", "loan", "personal"
]

# ----------------------------
# LOAD JSON
# ----------------------------
def load_ocr_json(path):
    with open(path, "r") as f:
        return json.load(f)


# ----------------------------
# FILTER LOW CONFIDENCE
# ----------------------------
def filter_noise(data, threshold=CONF_THRESHOLD):
    return [item for item in data if item["confidence"] >= threshold]


# ----------------------------
# SORT (Top → Bottom, Left → Right)
# ----------------------------
def sort_ocr(data):
    return sorted(data, key=lambda x: (x["bbox"][1], x["bbox"][0]))


# ----------------------------
# MERGE LINES
# ----------------------------
def merge_lines(data, y_threshold=Y_THRESHOLD):
    if not data:
        return []

    merged = []
    current = data[0].copy()

    for next_item in data[1:]:
        # check vertical closeness
        if abs(next_item["bbox"][1] - current["bbox"][3]) < y_threshold:
            current["text"] += " " + next_item["text"]

            # expand bbox
            current["bbox"][2] = max(current["bbox"][2], next_item["bbox"][2])
            current["bbox"][3] = next_item["bbox"][3]

            # avg confidence
            current["confidence"] = (current["confidence"] + next_item["confidence"]) / 2

        else:
            merged.append(current)
            current = next_item.copy()

    merged.append(current)
    return merged


# ----------------------------
# SECTION DETECTION
# ----------------------------
def detect_section(text):
    text_lower = text.lower()
    for key in SECTION_KEYS:
        if key in text_lower:
            return key
    return "general"


# ----------------------------
# CREATE CHUNKS
# ----------------------------
def create_chunks(data, page):
    chunks = []
    current_chunk = {
        "page": page,
        "section": None,
        "text": "",
        "items": [],
        "confidence": []
    }

    for item in data:
        section = detect_section(item["text"])

        # new section → new chunk
        if current_chunk["section"] != section:
            if current_chunk["text"]:
                chunks.append(format_chunk(current_chunk))

            current_chunk = {
                "page": page,
                "section": section,
                "text": item["text"],
                "items": [item],
                "confidence": [item["confidence"]]
            }
        else:
            current_chunk["text"] += " " + item["text"]
            current_chunk["items"].append(item)
            current_chunk["confidence"].append(item["confidence"])

    if current_chunk["text"]:
        chunks.append(format_chunk(current_chunk))

    return chunks


# ----------------------------
# FORMAT FINAL CHUNK
# ----------------------------
def format_chunk(chunk):
    avg_conf = sum(chunk["confidence"]) / len(chunk["confidence"])

    return {
        "text": chunk["text"],
        "metadata": {
            "page": chunk["page"],
            "section": chunk["section"],
            "avg_confidence": round(avg_conf, 3),
            "num_lines": len(chunk["items"])
        }
    }


# ----------------------------
# MAIN PIPELINE
# ----------------------------
def process_ocr_to_rag(input_json_path, output_path):
    ocr_data = load_ocr_json(input_json_path)

    all_chunks = []

    for page_name, items in ocr_data.items():
        page_number = int(page_name.split("_")[1].split(".")[0])

        # pipeline
        items = filter_noise(items)
        items = sort_ocr(items)
        items = merge_lines(items)

        chunks = create_chunks(items, page_number)
        all_chunks.extend(chunks)

    # save output
    with open(output_path, "w") as f:
        json.dump(all_chunks, f, indent=2)

    print(f"✅ RAG-ready data saved to {output_path}")


# ----------------------------
# RUN
# ----------------------------
if __name__ == "__main__":
    process_ocr_to_rag("ocr_spatial_data.json", "rag_output.json")

import requests
import os
import json
from google.auth.transport.requests import Request as GoogleRequest
from google.oauth2.credentials import Credentials

API_BASE = "https://photospicker.googleapis.com/v1"

def create_picker_session(creds):
    """Creates a Google Photos Picker session and returns the session name and picker URI."""
    headers = {
        "Authorization": f"Bearer {creds.token}",
        "Content-Type": "application/json"
    }
    # Payload must be an empty PickingSession object or omit it entirely
    payload = {}
    
    response = requests.post(f"{API_BASE}/sessions", headers=headers, json=payload)
    if response.status_code == 200:
        return response.json(), None
    else:
        try:
            error_data = response.json()
            error_msg = error_data.get("error", {}).get("message", "Unknown error occurred")
        except:
            error_msg = response.text or "Unknown error"
        return None, error_msg

def check_session_status(creds, session_id):
    """Checks if the user has finished picking photos in the session."""
    headers = {
        "Authorization": f"Bearer {creds.token}"
    }
    
    response = requests.get(f"{API_BASE}/sessions/{session_id}", headers=headers)
    if response.status_code == 200:
        data = response.json()
        # The field name is 'mediaItemsSet' (boolean)
        return data.get("mediaItemsSet", False)
    return False

def get_picked_media_items(creds, session_id):
    """Fetches media items picked during a session and downloads them with pagination support."""
    headers = {
        "Authorization": f"Bearer {creds.token}"
    }
    
    # Session ID for listing is the simple ID part
    simple_id = session_id.split('/')[-1]
    
    all_media_items = []
    next_page_token = None
    
    while True:
        url = f"{API_BASE}/mediaItems?sessionId={simple_id}"
        if next_page_token:
            url += f"&pageToken={next_page_token}"
            
        print(f"Fetching media items from: {url}")
        response = requests.get(url, headers=headers)
        
        if response.status_code != 200:
            print(f"Error fetching picked media items: {response.status_code} - {response.text}")
            break
            
        data = response.json()
        items = data.get("mediaItems", [])
        all_media_items.extend(items)
        
        next_page_token = data.get("nextPageToken")
        if not next_page_token:
            break
            
    print(f"Total media items found: {len(all_media_items)}")
    
    downloaded_count = 0
    os.makedirs("invoices", exist_ok=True)
    
    for item in all_media_items:
        # Picker API structure: mediaFile object contains baseUrl, filename, and mimeType
        media_file = item.get("mediaFile", {})
        base_url = media_file.get("baseUrl")
        original_filename = media_file.get("filename") or f"photo_{item.get('id', 'unknown')}"
        
        if base_url:
            print(f"Downloading {original_filename}...")
            # Append '=d' to baseUrl for original image download
            download_url = f"{base_url}=d"
            try:
                # Per Docs: Requests to baseUrl MUST include Authorization header
                img_response = requests.get(download_url, headers=headers)
                
                if img_response.status_code == 200:
                    # Clean filename to avoid issues
                    safe_filename = "".join([c for c in original_filename if c.isalnum() or c in "._- "]).strip()
                    filepath = os.path.join("invoices", f"picker_{item.get('id', '')[:8]}_{safe_filename}")
                    
                    with open(filepath, "wb") as f:
                        f.write(img_response.content)
                    print(f"Successfully saved to {filepath}")
                    downloaded_count += 1
                else:
                    print(f"Failed to download image: {img_response.status_code} - {img_response.text}")
            except Exception as e:
                print(f"Exception during download of {original_filename}: {e}")
        else:
            print(f"No baseUrl found for item {item.get('id')}")
            
    return downloaded_count

# Configure standalone app
app = FastAPI()
templates = Jinja2Templates(directory=os.path.join(os.path.dirname(__file__), "templates"))
TOKEN_FILE = os.path.join(os.path.dirname(__file__), "..", "token.json")
SCOPES = ['https://www.googleapis.com/auth/photospicker.mediaitems.readonly']

@app.get("/", response_class=HTMLResponse)
async def picker_home(request: Request):
    """Simple UI for the standalone picker."""
    is_authenticated = os.path.exists(TOKEN_FILE)
    if not is_authenticated:
        return "Please authenticate first using the main app."
    return templates.TemplateResponse("picker.html", {"request": request})

@app.post("/photos/start")
async def photos_start_endpoint():
    """Starts Google Photos Picker session."""
    if not os.path.exists(TOKEN_FILE):
        return {"error": "Authentication required"}
    
    creds = Credentials.from_authorized_user_file(TOKEN_FILE, SCOPES)
    if creds.expired and creds.refresh_token:
        creds.refresh(GoogleRequest())
    
    # Callback must match what's in your Google Console, using a dummy here
    # Since the REST API doesn't use redirects, this is mostly for the session creation
    session, error = create_picker_session(creds)
    
    if session and "pickerUri" in session:
        return {
            "status": "success", 
            "pickerUri": session["pickerUri"],
            "sessionId": session["id"] 
        }
    else:
        return {"error": error or "Failed to create picker session"}

@app.get("/photos/status/{sessionId:path}")
async def photos_status_endpoint(sessionId: str):
    """Checks completion status."""
    creds = Credentials.from_authorized_user_file(TOKEN_FILE, SCOPES)
    if creds.expired and creds.refresh_token:
        creds.refresh(GoogleRequest())
    
    is_done = check_session_status(creds, sessionId)
    return {"status": "success", "is_done": is_done}

        creds.refresh(GoogleRequest())
    
    count = get_picked_media_items(creds, sessionId)
    return {"status": "success", "count": count}

    <script>
        async function startPicker() {
            const statusDiv = document.getElementById('status');
            statusDiv.innerText = "Initializing session...";
            
            try {
                const response = await fetch('/photos/start', { method: 'POST' });
                const data = await response.json();
                
                if (data.status === 'success' && data.pickerUri) {
                    const pickerUrl = data.pickerUri + '/autoclose';
                    const sessionId = data.sessionId;
                    
                    window.open(pickerUrl, 'PickerPopup', 'width=600,height=700');
                    statusDiv.innerText = "Waiting for selection in popup...";
                    
                    const poller = setInterval(async () => {
                        const sResp = await fetch(`/photos/status/${sessionId}`);
                        const sData = await sResp.json();
                        
                        if (sData.is_done) {
                            clearInterval(poller);
                            statusDiv.innerText = "Downloading items...";
                            const cResp = await fetch(`/photos/complete/${sessionId}`, { method: 'POST' });
                            const cData = await cResp.json();
                            alert(`Successfully picked and downloaded ${cData.count} items!`);
                            statusDiv.innerText = "All items downloaded successfully.";
                        }
                    }, 3000);
                } else {
                    statusDiv.innerText = "Error: " + (data.error || "Unknown error");
                }
            } catch (e) {
                statusDiv.innerText = "Connection error: " + e.message;
            }
        }
    </script>
