drive.html Frontend

Minimal, unstyled HTML bridge logic to capture and transmit the scheduling timeline back to the background agent.

<!-- 1. Barebones User Interface Elements -->
<div>
    <label>Select Drive Folder: </label>
    <select id="drive-folder"></select>

    <label>Fetch Interval (Minutes): </label>
    <input type="number" id="drive-interval" value="60">

    <button onclick="saveDriveSettings()">Start Sync</button>
</div>

<!-- 2. Connection Logic -->
<script>
    // Step A: Request Google Drive folders from Backend to populate options
    async function loadDriveFolders() {
        const response = await fetch('/drive/folders');
        const data = await response.json();
        
        const select = document.getElementById('drive-folder');
        select.innerHTML = ''; 
        
        data.folders.forEach(folder => {
            select.innerHTML += `<option value="${folder.id}">${folder.name}</option>`;
        });
    }

    // Step B: Bundle Interval Timer + Target ID and POST to Backend Configuration
    async function saveDriveSettings() {
        const payload = {
            folder_id: document.getElementById('drive-folder').value,
            interval_minutes: document.getElementById('drive-interval').value
        };

        const response = await fetch('/drive/settings', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload)
        });
        
        const data = await response.json();
        if (data.status === 'success') {
            alert('Automation Saved! Fetching begins in the background.');
        }
    }

    // Auto-trigger folder extraction instantly on page load
    window.addEventListener('load', loadDriveFolders);
</script>

app.py Backend

The core structural backbone demonstrating how FastAPI sustains non-blocking infinite interval polling without breaking HTTP streams.

import os, json, asyncio
from fastapi import FastAPI, Request

app = FastAPI()
DRIVE_SETTINGS_FILE = "drive_settings.json"

# ======================================================================
# THE AUTOMATOR ENGINE
# ======================================================================
async def drive_auto_fetch_task():
    while True:
        # Await endlessly gracefully if settings aren't setup yet
        if not os.path.exists(DRIVE_SETTINGS_FILE):
            await asyncio.sleep(60) 
            continue
            
        # Dynamically read JSON config so restarts aren't required when updated via UI
        with open(DRIVE_SETTINGS_FILE, "r") as f:
            settings = json.load(f)
            
        folder_id = settings.get("folder_id")
        interval_minutes = int(settings.get("interval_minutes", 60))
        
        if folder_id:
            # ---> INSERT GOOGLE API CONNECTION + EXTRACTION HERE <---
            print(f"Triggering fetch loop for target Drive ID: {folder_id}")
            
        # CRITICAL YIELD: Relinquish core thread back to the uvicorn webserver
        await asyncio.sleep(interval_minutes * 60)

# Mount loop explicitly to Uvicorn startup lifecycle 
@app.on_event("startup")
async def startup_event():
    asyncio.create_task(drive_auto_fetch_task())


# ======================================================================
# API COMMUNICATION BRIDGES
# ======================================================================
@app.post("/drive/settings")
async def save_drive_settings(request: Request):
    """Write UI Form Configs strictly to Local Persistent Storage"""
    data = await request.json()
    with open(DRIVE_SETTINGS_FILE, "w") as f:
        json.dump({
            "folder_id": data.get("folder_id"), 
            "interval_minutes": data.get("interval_minutes", 60)
        }, f)
    return {"status": "success"}
