The core structural backbone demonstrating how FastAPI sustains non-blocking infinite interval polling without breaking HTTP streams.

import os, json, asyncio
from fastapi import FastAPI, Request

app = FastAPI()
DRIVE_SETTINGS_FILE = "drive_settings.json"

# ======================================================================
# THE AUTOMATOR ENGINE
# ======================================================================
async def drive_auto_fetch_task():
    while True:
        # Await endlessly gracefully if settings aren't setup yet
        if not os.path.exists(DRIVE_SETTINGS_FILE):
            await asyncio.sleep(60) 
            continue
            
        # Dynamically read JSON config so restarts aren't required when updated via UI
        with open(DRIVE_SETTINGS_FILE, "r") as f:
            settings = json.load(f)
            
        folder_id = settings.get("folder_id")
        interval_minutes = int(settings.get("interval_minutes", 60))
        
        if folder_id:
            # ---> INSERT GOOGLE API CONNECTION + EXTRACTION HERE <---
            print(f"Triggering fetch loop for target Drive ID: {folder_id}")
            
        # CRITICAL YIELD: Relinquish core thread back to the uvicorn webserver
        await asyncio.sleep(interval_minutes * 60)

# Mount loop explicitly to Uvicorn startup lifecycle 
@app.on_event("startup")
async def startup_event():
    asyncio.create_task(drive_auto_fetch_task())

pdf_manager.py Logic

Scanning engine that detects PDF protection layers and performs clean decryption using AES-256 standards.

import fitz # PyMuPDF

def check_pdf_protection(file_path):
    """Scan PDF for encryption state"""
    doc = fitz.open(file_path)
    is_protected = doc.is_encrypted
    # Return (is_locked, is_authenticated_with_empty_pass)
    return is_protected, doc.authenticate("") != 0

def unlock_pdf(file_path, password, output_path):
    """Decrypt PDF and save security-free version"""
    doc = fitz.open(file_path)
    if doc.authenticate(password):
        # Save with encryption=0 to remove all protection
        doc.save(output_path, encryption=0)
        return True, "Success"
    return False, "Invalid Password"

security_logic.js Logic

// 1. Toggle Password Visibility (The Eye Icon)
function togglePasswordVisibility() {
    const pwdInput = document.getElementById('pdf-password');
    const eyeIcon = document.getElementById('togglePassword');
    
    if (pwdInput.type === 'password') {
        pwdInput.type = 'text';
        eyeIcon.classList.replace('fa-eye', 'fa-eye-slash');
    } else {
        pwdInput.type = 'password';
        eyeIcon.classList.replace('fa-eye-slash', 'fa-eye');
    }
}

// 2. The Unlock automation bridge
async function submitPassword() {
    const password = document.getElementById('pdf-password').value;
    const formData = new FormData();
    formData.append('filename', currentFilename);
